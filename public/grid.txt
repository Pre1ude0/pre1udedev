function draw_grid () {
    fill('#000000');
    stroke('#000000');
    for (let i = 0; i < canv_width; i += 20) {
        if (i % 40 === 0 && i !== 0) {
            text(i, i, 10);
        }
        line(i, 0, i, canv_height);
    }
    for (let i = 0; i < canv_height; i += 20) {
        if (i % 40 === 0 && i !== 0) {
            text(i, 0, i);
        }
        line(0, i, canv_width, i);
    }
}
